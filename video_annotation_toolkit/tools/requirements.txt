# Preprocessing deps for tools/extract_mediapipe.py (NOT needed to run the web app).
# Install in a Python 3.9–3.12 environment:  pip install -r tools/requirements.txt
#
# Pins are deliberate — extract_mediapipe.py uses the legacy `mediapipe.solutions.pose` API:
#   * mediapipe 0.10.21 is the last release whose macOS/arm64 wheels still bundle
#     `mediapipe.solutions`. Newer wheels (e.g. 0.10.35) ship only `tasks/` + `modules/`,
#     so `import mediapipe.solutions` fails with ModuleNotFoundError.
#   * numpy must stay <2: mediapipe's compiled extensions are built against the numpy 1.x ABI.
mediapipe==0.10.21
numpy<2
opencv-python>=4.8
